\usepackage{xcolor}
\usepackage{listings}

\definecolor{lightgray}{rgb}{0.95, 0.95, 0.95}  % fondo más claro
\definecolor{keyword}{rgb}{0.7, 0.0, 0.0}        % color de las palabras clave (rojo)
\definecolor{comment}{rgb}{0.0, 0.5, 0.0}        % color de los comentarios (verde)
\definecolor{string}{rgb}{0.58, 0.0, 0.83}       % color de las cadenas (púrpura)
\definecolor{background}{rgb}{0.95, 0.95, 0.95}   % color de fondo más claro

\lstdefinestyle{custom}{
  backgroundcolor=\color{background},   % color de fondo gris más claro
  basicstyle=\ttfamily\small,           % tipo de fuente monoespaciada
  breaklines=true,                      % hacer que el texto se divida si es largo
  frame=single,                         % agregar un borde alrededor del bloque de código
  numbers=none,                         % sin numeración de líneas
  keywordstyle=\color{keyword}\bfseries, % palabras clave en rojo y en negrita
  commentstyle=\color{comment},         % comentarios en verde
  stringstyle=\color{string},           % cadenas en púrpura
}

\lstset{style=custom}
